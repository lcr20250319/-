def printing_Kfold_scores(x_train_data,y_train_data):
    fold = KFold(5,shuffle=False)
    
    
    
    c_param_range = [0.01,0.1,1,10,100]

    
    results_table = pd.DataFrame(index=range(len(c_param_range),2),columns = ['C_parameter','Mean recall score'])
    results_table['C_parameter'] = c_param_range

    j = 0 
    for c_param in c_param_range:
        print('-------------------------------------------')
        print('正则化惩罚力度: ', c_param)
        print('-------------------------------------------')
        print('')

        recall_accs = []

        for iteration, indices in enumerate(fold.split(x_train_data),start=1): 
            
                        
            
            
            
            
                
            
            
    
            
            lr = LogisticRegression(C = c_param, penalty = 'l1',solver='liblinear')
    
            
            
            lr.fit(x_train_data.iloc[indices[0],:],y_train_data.iloc[indices[0],:].values.ravel())
    
            
            y_pred_undersample = lr.predict(x_train_data.iloc[indices[1],:].values)
            
    
            
            
            recall_acc = recall_score(y_train_data.iloc[indices[1],:].values,y_pred_undersample)
            
            recall_accs.append(recall_acc)
            print('Iteration',iteration,':召回率=',recall_acc)

        
        results_table.loc[j,'Mean recall score'] =np.mean(recall_accs)
        
        j +=1
        print("")
        print("平均召回率:",np.mean(recall_accs))
        print("")

    
    best_c = results_table.loc[results_table['Mean recall score'].astype(float).idxmax()]['C_parameter']
    
    print('*********************************************************************************')
    print('效果最好的模型所选参数 = ', best_c)
    print('*********************************************************************************')
    
    return best_c

def plot_confusion_matrix(cm,classes,title="Confusion matrix",
                          cmap=plt.cm.Blues):
    plt.imshow(cm,cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(classes))

    
    plt.xticks(tick_marks, classes,rotation=0)
    plt.yticks(tick_marks, classes)  

    
    thresh = cm.max()/2.
    for i,j in itertools.product(range(cm.shape[0]),range(cm.shape[1])):
        
        plt.text(j,i,cm[i,j],
                 horizontalalignment="center",
                 color="white" if cm[i,j] > thresh else "black")
        
        
        

    plt.ylabel('True label')
    plt.xlabel('Predicted label')

上述结果解释:随着阈值增大,召回率降低了,因为判断异常的标准提高了.当阈值为0.1-0.3时,召回率为1,所用样本都为异常,这样的模型是没有意义;当阈值为0.4时,召回率高但是假阳性太高,不用;当阈值为0.6时,对比0.5,漏检变多,假阳性变少,具体选哪个要看实际指标;当阈值为0.7-0.9时,召回率太低,模型不可取.

下采样方案有局限性,假阳性高,下面采用过采样方案

(过采样:正负的数据量不平衡,少向多的一方的增加数量,达到平衡. 例:正有100,负有10,把负数量调节为100. 下采样:正负的数据量不平衡,多向少的一方的减少数量,达到平衡. 例:正有100,负有10,把正数量调节为10.)

过采样方案如何生成策略,在两个样本连线上随机生成新样本.

步骤: 1.判断那部分是少数样本; 2.计算样本间距离 3.将样本距离从小到大排序 4.确定一个倍率k=i,选最小的i个距离d 5.x_new=x+Random(0,1)*d
